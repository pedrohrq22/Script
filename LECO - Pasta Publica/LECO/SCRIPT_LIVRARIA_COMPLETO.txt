(COD_PEDIDO DECIMAL(38),
COD_CLIENTE DECIMAL(38) NOT NULL,
VALOR_TOTAL DECIMAL(5,2) NOT NULL,

);

--Criando a tabela ITEM_PEDIDO
CREATE TABLE ITEM_PEDIDO
(COD_ITEM_PEDIDO DECIMAL(38),
COD_PEDIDO DECIMAL(38) NOT NULL,
COD_LIVRO DECIMAL(38) NOT NULL,
QUANTIDADE DECIMAL(4) DEFAULT 1,
CONSTRAINT PK_ITEM_PEDIDO PRIMARY KEY (COD_ITEM_PEDIDO),
CONSTRAINT FK_PEDIDO FOREIGN KEY (COD_PEDIDO) REFERENCES PEDIDO (COD_PEDIDO),
CONSTRAINT FK_LIVRO_PEDIDO FOREIGN KEY (COD_LIVRO) REFERENCES LIVRO (COD_LIVRO)
);
